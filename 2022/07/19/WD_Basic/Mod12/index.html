functionObject: water_in_domain

  • volume integral of the field variables $\alpha$ in all the domain

      function 
  {

  water_in_domain
  {
      type            volFieldValue;
      functionObjectLibs ("libfieldFunctionObjects.so");

      enabled         true;
      log             true;

      //writeControl    writeTime;
      writeControl   timeStep;
      writeInterval  1;

      writeFields     false;
      writeLocation   false;    

      regionType      all;

      operation       volIntegrate;   //volume integral
      fields
      (
          alpha.water
      );
  }

  }
