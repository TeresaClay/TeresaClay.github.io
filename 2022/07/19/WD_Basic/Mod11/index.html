      fvSchemes

      • second order accurate, but oscillatory

        ddtSchemes
{
    default         backward;
}

gradSchemes
{
    default         Gauss linear;
    grad(p)         Gauss linear;
}

divSchemes
{
    default         none;
    div(phi,U)      Gauss linear;

    div((nuEff*dev2(T(grad(U))))) Gauss linear;
}

laplacianSchemes
{
    //default         Gauss linear orthogonal;
    default         Gauss linear limited 1;     // limited 1 corrections to handle mesh non-orthogonality and non-uniformity
}

interpolationSchemes
{
    default         linear;
}

snGradSchemes
{
    //default         orthogonal;
    default         limited 1;
}

        fvSolution

      • additional sub-dictionaries PISO, PIMPLE, SIMPLE and relaxationFactors

      • in this case, only PISO correction, no orthogonal corrections

        solvers
{
    p
    {
        solver          PCG;
        preconditioner  DIC;
        tolerance       1e-06;
        relTol          0;

/*
        solver           GAMG;
        tolerance        1e-6;
        relTol           0.0;
        smoother         GaussSeidel;
        nPreSweeps       0;
        nPostSweeps      2;
        cacheAgglomeration on;
        agglomerator     faceAreaPair;
        nCellsInCoarsestLevel 100;
        mergeLevels      1;
*/

    }

    pFinal            // putting more computational effort in the last iteration (but for this case the same for p and pFinal)
    {
        $p;
        relTol          0;
    }

    U
    {
        solver          smoothSolver;
        smoother        symGaussSeidel;
        tolerance       1e-08;
        relTol          0;
    }
}

PISO                                // pressure-velocity coupling
{
    //consistent         yes;
    //momentumPredictor   yes;

    nCorrectors     1;             // at leat one PISO l_op 
    nNonOrthogonalCorrectors 1;
    pRefCell        0;
    pRefValue       0;
}

relaxationFactors
{
    fields
    {
        ".*" 0.9;
    }
    equations
    {
        ".*" 0.9;
    }
}
